import { GET } from '@z-fetch/fetch';

async function getPosts() {
  const { data, error } = await GET('https://jsonplaceholder.typicode.com/posts');
  if (data) {
    console.log('Posts:', data);
  } else {
    console.error('Error fetching posts:', error);
  }
}

getPosts();
